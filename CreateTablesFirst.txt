CREATE DATABASE IF NOT EXISTS csci_467;

CREATE TABLE User(user_id INT AUTO_INCREMENT PRIMARY KEY, Uusername VARCHAR(30) PRIMARY KEY, Upassword VARCHAR(30), address VARCHAR(30), commission VARCHAR(30)); 

CREATE TABLE quote(quote_id INT AUTO_INCREMENT PRIMARY KEY, customer VARCHAR (255), address VARCHAR(255), email VARCHAR(25), line_item VARCHAR(25), line_item_price VARCHAR(25), secret_note VARCHAR(255), amount INT);
